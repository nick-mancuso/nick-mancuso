78      /**
79       * The ServerHello handshake message.
80       */
81      static final class ServerHelloMessage extends HandshakeMessage {
82          final ProtocolVersion           serverVersion;      // TLS 1.3 legacy
83          final RandomCookie              serverRandom;
84          final SessionId                 sessionId;          // TLS 1.3 legacy
85          final CipherSuite               cipherSuite;
86          final byte                      compressionMethod;  // TLS 1.3 legacy
87          final SSLExtensions             extensions;
88  
89          // The HelloRetryRequest producer needs to use the ClientHello message
90          // for cookie generation.  Please don't use this field for other
91          // purpose unless it is really necessary.
92          final ClientHelloMessage        clientHello;
93  
94          // Reserved for HelloRetryRequest consumer.  Please don't use this
95          // field for other purpose unless it is really necessary.
96          final ByteBuffer                handshakeRecord;
97  
98          ServerHelloMessage(HandshakeContext context,
99                  ProtocolVersion serverVersion, SessionId sessionId,
100                 CipherSuite cipherSuite, RandomCookie serverRandom,
101                 ClientHelloMessage clientHello) {
102             super(context);
103 
104             this.serverVersion = serverVersion;
105             this.serverRandom = serverRandom;
106             this.sessionId = sessionId;
107             this.cipherSuite = cipherSuite;
108             this.compressionMethod = 0x00;      // Don't support compression.
109             this.extensions = new SSLExtensions(this);
110 
111             // Reserve the ClientHello message for cookie generation.
112             this.clientHello = clientHello;
113 
114             // The handshakeRecord field is used for HelloRetryRequest consumer
115             // only.  It's fine to set it to null for generating side of the
116             // ServerHello/HelloRetryRequest message.
117             this.handshakeRecord = null;
118         }
119 
120         ServerHelloMessage(HandshakeContext context,
121                 ByteBuffer m) throws IOException {
122             super(context);
123 
124             // Reserve for HelloRetryRequest consumer if needed.
125             this.handshakeRecord = m.duplicate();
126 
127             byte major = m.get();
128             byte minor = m.get();
129             this.serverVersion = ProtocolVersion.valueOf(major, minor);
130             if (this.serverVersion == null) {
131                 // The client should only request for known protocol versions.
132                 throw context.conContext.fatal(Alert.PROTOCOL_VERSION,
133                     "Unsupported protocol version: " +
134                     ProtocolVersion.nameOf(major, minor));
135             }
136 
137             this.serverRandom = new RandomCookie(m);
138             this.sessionId = new SessionId(Record.getBytes8(m));
139             try {
140                 sessionId.checkLength(serverVersion.id);
141             } catch (SSLProtocolException ex) {
142                 throw handshakeContext.conContext.fatal(
143                         Alert.ILLEGAL_PARAMETER, ex);
144             }
145 
146             int cipherSuiteId = Record.getInt16(m);
147             this.cipherSuite = CipherSuite.valueOf(cipherSuiteId);
148             if (cipherSuite == null || !context.isNegotiable(cipherSuite)) {
149                 throw context.conContext.fatal(Alert.ILLEGAL_PARAMETER,
150                     "Server selected improper ciphersuite " +
151                     CipherSuite.nameOf(cipherSuiteId));
152             }
153 
154             this.compressionMethod = m.get();
155             if (compressionMethod != 0) {
156                 throw context.conContext.fatal(Alert.ILLEGAL_PARAMETER,
157                     "compression type not supported, " + compressionMethod);
158             }
159 
160             SSLExtension[] supportedExtensions;
161             if (serverRandom.isHelloRetryRequest()) {
162                 supportedExtensions = context.sslConfig.getEnabledExtensions(
163                             SSLHandshake.HELLO_RETRY_REQUEST);
164             } else {
165                 supportedExtensions = context.sslConfig.getEnabledExtensions(
166                             SSLHandshake.SERVER_HELLO);
167             }
168 
169             if (m.hasRemaining()) {
170                 this.extensions =
171                     new SSLExtensions(this, m, supportedExtensions);
172             } else {
173                 this.extensions = new SSLExtensions(this);
174             }
175 
176             // The clientHello field is used for HelloRetryRequest producer
177             // only.  It's fine to set it to null for receiving side of
178             // ServerHello/HelloRetryRequest message.
179             this.clientHello = null;        // not used, let it be null;
180         }
181 
182         @Override
183         public SSLHandshake handshakeType() {
184             return serverRandom.isHelloRetryRequest() ?
185                 SSLHandshake.HELLO_RETRY_REQUEST : SSLHandshake.SERVER_HELLO;
186         }
187 
188         @Override
189         public int messageLength() {
190             // almost fixed header size, except session ID and extensions:
191             //      major + minor = 2
192             //      random = 32
193             //      session ID len field = 1
194             //      cipher suite = 2
195             //      compression = 1
196             //      extensions: if present, 2 + length of extensions
197             // In TLS 1.3, use of certain extensions is mandatory.
198             return 38 + sessionId.length() + extensions.length();
199         }
200 
201         @Override
202         public void send(HandshakeOutStream hos) throws IOException {
203             hos.putInt8(serverVersion.major);
204             hos.putInt8(serverVersion.minor);
205             hos.write(serverRandom.randomBytes);
206             hos.putBytes8(sessionId.getId());
207             hos.putInt8((cipherSuite.id >> 8) & 0xFF);
208             hos.putInt8(cipherSuite.id & 0xff);
209             hos.putInt8(compressionMethod);
210 
211             extensions.send(hos);           // In TLS 1.3, use of certain
212                                             // extensions are mandatory.
213         }
214 
215         @Override
216         public String toString() {
217             MessageFormat messageFormat = new MessageFormat(
218                     """
219                             "{0}": '{'
220                               "server version"      : "{1}",
221                               "random"              : "{2}",
222                               "session id"          : "{3}",
223                               "cipher suite"        : "{4}",
224                               "compression methods" : "{5}",
225                               "extensions"          : [
226                             {6}
227                               ]
228                             '}'""",
229                 Locale.ENGLISH);
230             Object[] messageFields = {
231                 serverRandom.isHelloRetryRequest() ?
232                     "HelloRetryRequest" : "ServerHello",
233                 serverVersion.name,
234                 Utilities.toHexString(serverRandom.randomBytes),
235                 sessionId.toString(),
236                 cipherSuite.name + "(" + Utilities.byte16HexString(cipherSuite.id) + ")",
237                 HexFormat.of().toHexDigits(compressionMethod),
238                 Utilities.indent(extensions.toString(), "    ")
239             };
240 
241             return messageFormat.format(messageFields);
242         }
243     }

248     private static final class T12ServerHelloProducer
249             implements HandshakeProducer {
250 
251         // Prevent instantiation of this class.
252         private T12ServerHelloProducer() {
253             // blank
254         }
255 
256         @Override
257         public byte[] produce(ConnectionContext context,
258                 HandshakeMessage message) throws IOException {
259             // The producing happens in server side only.
260             ServerHandshakeContext shc = (ServerHandshakeContext)context;
261             ClientHelloMessage clientHello = (ClientHelloMessage)message;
262 
263             // If client hasn't specified a session we can resume, start a
264             // new one and choose its cipher suite and compression options,
265             // unless new session creation is disabled for this connection!
266             if (!shc.isResumption || shc.resumingSession == null) {
267                 if (!shc.sslConfig.enableSessionCreation) {
268                     throw new SSLException(
269                         "Not resumption, and no new session is allowed");
270                 }
271 
272                 if (shc.localSupportedSignAlgs == null) {
273                     shc.localSupportedSignAlgs =
274                         SignatureScheme.getSupportedAlgorithms(
275                                 shc.sslConfig,
276                                 shc.algorithmConstraints, shc.activeProtocols);
277                 }
278 
279                 SSLSessionImpl session =
280                         new SSLSessionImpl(shc, CipherSuite.C_NULL);
281                 session.setMaximumPacketSize(shc.sslConfig.maximumPacketSize);
282                 shc.handshakeSession = session;
283 
284                 // consider the handshake extension impact
285                 SSLExtension[] enabledExtensions =
286                         shc.sslConfig.getEnabledExtensions(
287                             SSLHandshake.CLIENT_HELLO, shc.negotiatedProtocol);
288                 clientHello.extensions.consumeOnTrade(shc, enabledExtensions);
289 
290                 // negotiate the cipher suite.
291                 KeyExchangeProperties credentials =
292                         chooseCipherSuite(shc, clientHello);
293                 if (credentials == null) {
294                     throw shc.conContext.fatal(Alert.HANDSHAKE_FAILURE,
295                             "no cipher suites in common");
296                 }
297                 shc.negotiatedCipherSuite = credentials.cipherSuite;
298                 shc.handshakeKeyExchange = credentials.keyExchange;
299                 shc.handshakeSession.setSuite(credentials.cipherSuite);
300                 shc.handshakePossessions.addAll(
301                         Arrays.asList(credentials.possessions));
302                 shc.handshakeHash.determine(
303                         shc.negotiatedProtocol, shc.negotiatedCipherSuite);
304 
305                 // Check the incoming OCSP stapling extensions and attempt
306                 // to get responses.  If the resulting stapleParams is non-null,
307                 // it implies that stapling is enabled on the server side.
308                 shc.stapleParams = StatusResponseManager.processStapling(shc);
309                 shc.staplingActive = (shc.stapleParams != null);
310 
311                 // update the responders
312                 SSLKeyExchange ke = credentials.keyExchange;
313                 if (ke != null) {
314                     for (Map.Entry<Byte, HandshakeProducer> me :
315                             ke.getHandshakeProducers(shc)) {
316                         shc.handshakeProducers.put(
317                                 me.getKey(), me.getValue());
318                     }
319                 }
320 
321                 if ((ke != null) &&
322                         (shc.sslConfig.clientAuthType !=
323                                 ClientAuthType.CLIENT_AUTH_NONE) &&
324                         !shc.negotiatedCipherSuite.isAnonymous()) {
325                     for (SSLHandshake hs :
326                             ke.getRelatedHandshakers(shc)) {
327                         if (hs == SSLHandshake.CERTIFICATE) {
328                             shc.handshakeProducers.put(
329                                     SSLHandshake.CERTIFICATE_REQUEST.id,
330                                     SSLHandshake.CERTIFICATE_REQUEST);
331                             break;
332                         }
333                     }
334                 }
335                 shc.handshakeProducers.put(SSLHandshake.SERVER_HELLO_DONE.id,
336                         SSLHandshake.SERVER_HELLO_DONE);
337             } else {
338                 // stateless and use the client session id (RFC 5077 3.4)
339                 if (shc.statelessResumption) {
340                     shc.resumingSession = new SSLSessionImpl(shc.resumingSession,
341                             (clientHello.sessionId.length() == 0) ?
342                                     new SessionId(true,
343                                             shc.sslContext.getSecureRandom()) :
344                                     new SessionId(clientHello.sessionId.getId())
345                     );
346                 }
347                 shc.handshakeSession = shc.resumingSession;
348                 shc.negotiatedProtocol =
349                         shc.resumingSession.getProtocolVersion();
350                 shc.negotiatedCipherSuite = shc.resumingSession.getSuite();
351                 shc.handshakeHash.determine(
352                         shc.negotiatedProtocol, shc.negotiatedCipherSuite);
353             }
354 
355             // Generate the ServerHello handshake message.
356             ServerHelloMessage shm = new ServerHelloMessage(shc,
357                     shc.negotiatedProtocol,
358                     shc.handshakeSession.getSessionId(),
359                     shc.negotiatedCipherSuite,
360                     new RandomCookie(shc),
361                     clientHello);
362             shc.serverHelloRandom = shm.serverRandom;
363 
364             // Produce extensions for ServerHello handshake message.
365             SSLExtension[] serverHelloExtensions =
366                 shc.sslConfig.getEnabledExtensions(
367                         SSLHandshake.SERVER_HELLO, shc.negotiatedProtocol);
368             shm.extensions.produce(shc, serverHelloExtensions);
369             if (SSLLogger.isOn && SSLLogger.isOn("ssl,handshake")) {
370                 SSLLogger.fine("Produced ServerHello handshake message", shm);
371             }
372 
373             // Output the handshake message.
374             shm.write(shc.handshakeOutput);
375             shc.handshakeOutput.flush();
376 
377             if (shc.isResumption && shc.resumingSession != null) {
378                 SSLTrafficKeyDerivation kdg =
379                     SSLTrafficKeyDerivation.valueOf(shc.negotiatedProtocol);
380                 if (kdg == null) {
381                     // unlikely
382                     throw shc.conContext.fatal(Alert.INTERNAL_ERROR,
383                             "Not supported key derivation: " +
384                             shc.negotiatedProtocol);
385                 } else {
386                     shc.handshakeKeyDerivation = kdg.createKeyDerivation(
387                             shc, shc.resumingSession.getMasterSecret());
388                 }
389 
390                 // update the responders
391                 shc.handshakeProducers.put(SSLHandshake.FINISHED.id,
392                         SSLHandshake.FINISHED);
393             }
394 
395             // The handshake message has been delivered.
396             return null;
397         }

399         private static KeyExchangeProperties chooseCipherSuite(
400                 ServerHandshakeContext shc,
401                 ClientHelloMessage clientHello) throws IOException {
402             List<CipherSuite> preferred;
403             List<CipherSuite> proposed;
404             if (shc.sslConfig.preferLocalCipherSuites) {
405                 preferred = shc.activeCipherSuites;
406                 proposed = clientHello.cipherSuites;
407             } else {
408                 preferred = clientHello.cipherSuites;
409                 proposed = shc.activeCipherSuites;
410             }
411 
412             List<CipherSuite> legacySuites = new LinkedList<>();
413             for (CipherSuite cs : preferred) {
414                 if (!HandshakeContext.isNegotiable(
415                         proposed, shc.negotiatedProtocol, cs)) {
416                     continue;
417                 }
418 
419                 if (shc.sslConfig.clientAuthType ==
420                         ClientAuthType.CLIENT_AUTH_REQUIRED) {
421                     if ((cs.keyExchange == KeyExchange.K_DH_ANON) ||
422                         (cs.keyExchange == KeyExchange.K_ECDH_ANON)) {
423                         continue;
424                     }
425                 }
426 
427                 SSLKeyExchange ke = SSLKeyExchange.valueOf(
428                         cs.keyExchange, shc.negotiatedProtocol);
429                 if (ke == null) {
430                     continue;
431                 }
432                 if (!ServerHandshakeContext.legacyAlgorithmConstraints.permits(
433                         EnumSet.of(CryptoPrimitive.KEY_AGREEMENT), cs.name, null)) {
434                     legacySuites.add(cs);
435                     continue;
436                 }
437 
438                 SSLPossession[] hcds = ke.createPossessions(shc);
439                 if ((hcds == null) || (hcds.length == 0)) {
440                     continue;
441                 }
442 
443                 // The cipher suite has been negotiated.
444                 if (SSLLogger.isOn && SSLLogger.isOn("ssl,handshake")) {
445                     SSLLogger.fine("use cipher suite " + cs.name);
446                 }
447 
448                 return new KeyExchangeProperties(cs, ke, hcds);
449             }
450 
451             for (CipherSuite cs : legacySuites) {
452                 SSLKeyExchange ke = SSLKeyExchange.valueOf(
453                         cs.keyExchange,  shc.negotiatedProtocol);
454                 if (ke != null) {
455                     SSLPossession[] hcds = ke.createPossessions(shc);
456                     if ((hcds != null) && (hcds.length != 0)) {
457                         if (SSLLogger.isOn && SSLLogger.isOn("ssl,handshake")) {
458                             SSLLogger.warning(
459                                 "use legacy cipher suite " + cs.name);
460                         }
461                         return new KeyExchangeProperties(cs, ke, hcds);
462                     }
463                 }
464             }
465 
466             throw shc.conContext.fatal(Alert.HANDSHAKE_FAILURE,
467                     "no cipher suites in common");
468         }

470         private static final class KeyExchangeProperties {
471             final CipherSuite cipherSuite;
472             final SSLKeyExchange keyExchange;
473             final SSLPossession[] possessions;
474 
475             private KeyExchangeProperties(CipherSuite cipherSuite,
476                     SSLKeyExchange keyExchange, SSLPossession[] possessions) {
477                 this.cipherSuite = cipherSuite;
478                 this.keyExchange = keyExchange;
479                 this.possessions = possessions;
480             }
481         }

484     /**
485      * The "ServerHello" handshake message producer.
486      */
487     private static final
488             class T13ServerHelloProducer implements HandshakeProducer {
489         // Prevent instantiation of this class.
490         private T13ServerHelloProducer() {
491             // blank
492         }
493 
494         @Override
495         public byte[] produce(ConnectionContext context,
496                 HandshakeMessage message) throws IOException {
497             // The producing happens in server side only.
498             ServerHandshakeContext shc = (ServerHandshakeContext)context;
499             ClientHelloMessage clientHello = (ClientHelloMessage)message;
500 
501             SSLSessionContextImpl sessionCache = (SSLSessionContextImpl)
502                     shc.sslContext.engineGetServerSessionContext();
503 
504             // If client hasn't specified a session we can resume, start a
505             // new one and choose its cipher suite and compression options,
506             // unless new session creation is disabled for this connection!
507             if (!shc.isResumption || shc.resumingSession == null) {
508                 if (!shc.sslConfig.enableSessionCreation) {
509                     throw new SSLException(
510                         "Not resumption, and no new session is allowed");
511                 }
512 
513                 if (shc.localSupportedSignAlgs == null) {
514                     shc.localSupportedSignAlgs =
515                         SignatureScheme.getSupportedAlgorithms(
516                                 shc.sslConfig,
517                                 shc.algorithmConstraints, shc.activeProtocols);
518                 }
519 
520                 SSLSessionImpl session =
521                         new SSLSessionImpl(shc, CipherSuite.C_NULL);
522                 session.setMaximumPacketSize(shc.sslConfig.maximumPacketSize);
523                 shc.handshakeSession = session;
524 
525                 // consider the handshake extension impact
526                 SSLExtension[] enabledExtensions =
527                         shc.sslConfig.getEnabledExtensions(
528                             SSLHandshake.CLIENT_HELLO, shc.negotiatedProtocol);
529                 clientHello.extensions.consumeOnTrade(shc, enabledExtensions);
530 
531                 // negotiate the cipher suite.
532                 CipherSuite cipherSuite = chooseCipherSuite(shc, clientHello);
533                 if (cipherSuite == null) {
534                     throw shc.conContext.fatal(Alert.HANDSHAKE_FAILURE,
535                             "no cipher suites in common");
536                 }
537                 shc.negotiatedCipherSuite = cipherSuite;
538                 shc.handshakeSession.setSuite(cipherSuite);
539                 shc.handshakeHash.determine(
540                         shc.negotiatedProtocol, shc.negotiatedCipherSuite);
541             } else {
542                 shc.handshakeSession = shc.resumingSession;
543 
544                 // consider the handshake extension impact
545                 SSLExtension[] enabledExtensions =
546                 shc.sslConfig.getEnabledExtensions(
547                 SSLHandshake.CLIENT_HELLO, shc.negotiatedProtocol);
548                 clientHello.extensions.consumeOnTrade(shc, enabledExtensions);
549 
550                 shc.negotiatedProtocol =
551                         shc.resumingSession.getProtocolVersion();
552                 shc.negotiatedCipherSuite = shc.resumingSession.getSuite();
553                 shc.handshakeHash.determine(
554                         shc.negotiatedProtocol, shc.negotiatedCipherSuite);
555 
556                 setUpPskKD(shc,
557                         shc.resumingSession.consumePreSharedKey());
558             }
559 
560             // update the responders
561             shc.handshakeProducers.put(SSLHandshake.ENCRYPTED_EXTENSIONS.id,
562                     SSLHandshake.ENCRYPTED_EXTENSIONS);
563             shc.handshakeProducers.put(SSLHandshake.FINISHED.id,
564                     SSLHandshake.FINISHED);
565 
566             // Generate the ServerHello handshake message.
567             ServerHelloMessage shm = new ServerHelloMessage(shc,
568                     ProtocolVersion.TLS12,      // use legacy version
569                     clientHello.sessionId,      // echo back
570                     shc.negotiatedCipherSuite,
571                     new RandomCookie(shc),
572                     clientHello);
573             shc.serverHelloRandom = shm.serverRandom;
574 
575             // Produce extensions for ServerHello handshake message.
576             SSLExtension[] serverHelloExtensions =
577                     shc.sslConfig.getEnabledExtensions(
578                         SSLHandshake.SERVER_HELLO, shc.negotiatedProtocol);
579             shm.extensions.produce(shc, serverHelloExtensions);
580             if (SSLLogger.isOn && SSLLogger.isOn("ssl,handshake")) {
581                 SSLLogger.fine("Produced ServerHello handshake message", shm);
582             }
583 
584             // Output the handshake message.
585             shm.write(shc.handshakeOutput);
586             shc.handshakeOutput.flush();
587 
588             // Change client/server handshake traffic secrets.
589             // Refresh handshake hash
590             shc.handshakeHash.update();
591 
592             // Change client/server handshake traffic secrets.
593             SSLKeyExchange ke = shc.handshakeKeyExchange;
594             if (ke == null) {
595                 // unlikely
596                 throw shc.conContext.fatal(Alert.INTERNAL_ERROR,
597                         "Not negotiated key shares");
598             }
599 
600             SSLKeyDerivation handshakeKD = ke.createKeyDerivation(shc);
601             SecretKey handshakeSecret = handshakeKD.deriveKey(
602                     "TlsHandshakeSecret", null);
603 
604             SSLTrafficKeyDerivation kdg =
605                 SSLTrafficKeyDerivation.valueOf(shc.negotiatedProtocol);
606             if (kdg == null) {
607                 // unlikely
608                 throw shc.conContext.fatal(Alert.INTERNAL_ERROR,
609                         "Not supported key derivation: " +
610                         shc.negotiatedProtocol);
611             }
612 
613             SSLKeyDerivation kd =
614                     new SSLSecretDerivation(shc, handshakeSecret);
615 
616             // update the handshake traffic read keys.
617             SecretKey readSecret = kd.deriveKey(
618                     "TlsClientHandshakeTrafficSecret", null);
619             SSLKeyDerivation readKD =
620                     kdg.createKeyDerivation(shc, readSecret);
621             SecretKey readKey = readKD.deriveKey(
622                     "TlsKey", null);
623             SecretKey readIvSecret = readKD.deriveKey(
624                     "TlsIv", null);
625             IvParameterSpec readIv =
626                     new IvParameterSpec(readIvSecret.getEncoded());
627             SSLReadCipher readCipher;
628             try {
629                 readCipher =
630                     shc.negotiatedCipherSuite.bulkCipher.createReadCipher(
631                         Authenticator.valueOf(shc.negotiatedProtocol),
632                         shc.negotiatedProtocol, readKey, readIv,
633                         shc.sslContext.getSecureRandom());
634             } catch (GeneralSecurityException gse) {
635                 // unlikely
636                 throw shc.conContext.fatal(Alert.HANDSHAKE_FAILURE,
637                         "Missing cipher algorithm", gse);
638             }
639 
640             if (readCipher == null) {
641                 throw shc.conContext.fatal(Alert.ILLEGAL_PARAMETER,
642                     "Illegal cipher suite (" + shc.negotiatedCipherSuite +
643                     ") and protocol version (" + shc.negotiatedProtocol +
644                     ")");
645             }
646 
647             shc.baseReadSecret = readSecret;
648             shc.conContext.inputRecord.changeReadCiphers(readCipher);
649 
650             // update the handshake traffic write secret.
651             SecretKey writeSecret = kd.deriveKey(
652                     "TlsServerHandshakeTrafficSecret", null);
653             SSLKeyDerivation writeKD =
654                     kdg.createKeyDerivation(shc, writeSecret);
655             SecretKey writeKey = writeKD.deriveKey(
656                     "TlsKey", null);
657             SecretKey writeIvSecret = writeKD.deriveKey(
658                     "TlsIv", null);
659             IvParameterSpec writeIv =
660                     new IvParameterSpec(writeIvSecret.getEncoded());
661             SSLWriteCipher writeCipher;
662             try {
663                 writeCipher =
664                     shc.negotiatedCipherSuite.bulkCipher.createWriteCipher(
665                         Authenticator.valueOf(shc.negotiatedProtocol),
666                         shc.negotiatedProtocol, writeKey, writeIv,
667                         shc.sslContext.getSecureRandom());
668             } catch (GeneralSecurityException gse) {
669                 // unlikely
670                 throw shc.conContext.fatal(Alert.HANDSHAKE_FAILURE,
671                         "Missing cipher algorithm", gse);
672             }
673 
674             if (writeCipher == null) {
675                 throw shc.conContext.fatal(Alert.ILLEGAL_PARAMETER,
676                     "Illegal cipher suite (" + shc.negotiatedCipherSuite +
677                     ") and protocol version (" + shc.negotiatedProtocol +
678                     ")");
679             }
680 
681             shc.baseWriteSecret = writeSecret;
682             shc.conContext.outputRecord.changeWriteCiphers(
683                     writeCipher, (clientHello.sessionId.length() != 0));
684 
685             // Update the context for master key derivation.
686             shc.handshakeKeyDerivation = kd;
687 
688             // Check if the server supports stateless resumption
689             if (sessionCache.statelessEnabled()) {
690                 shc.statelessResumption = true;
691             }
692 
693             // The handshake message has been delivered.
694             return null;
695         }

697         private static CipherSuite chooseCipherSuite(
698                 ServerHandshakeContext shc,
699                 ClientHelloMessage clientHello) {
700             List<CipherSuite> preferred;
701             List<CipherSuite> proposed;
702             if (shc.sslConfig.preferLocalCipherSuites) {
703                 preferred = shc.activeCipherSuites;
704                 proposed = clientHello.cipherSuites;
705             } else {
706                 preferred = clientHello.cipherSuites;
707                 proposed = shc.activeCipherSuites;
708             }
709 
710             CipherSuite legacySuite = null;
711             AlgorithmConstraints legacyConstraints =
712                     ServerHandshakeContext.legacyAlgorithmConstraints;
713             for (CipherSuite cs : preferred) {
714                 if (!HandshakeContext.isNegotiable(
715                         proposed, shc.negotiatedProtocol, cs)) {
716                     continue;
717                 }
718 
719                 if ((legacySuite == null) &&
720                         !legacyConstraints.permits(
721                                 EnumSet.of(CryptoPrimitive.KEY_AGREEMENT),
722                                 cs.name, null)) {
723                     legacySuite = cs;
724                     continue;
725                 }
726 
727                 // The cipher suite has been negotiated.
728                 if (SSLLogger.isOn && SSLLogger.isOn("ssl,handshake")) {
729                     SSLLogger.fine("use cipher suite " + cs.name);
730                 }
731                 return cs;
732             }
733 
734             if (legacySuite != null) {
735                 if (SSLLogger.isOn && SSLLogger.isOn("ssl,handshake")) {
736                     SSLLogger.warning(
737                             "use legacy cipher suite " + legacySuite.name);
738                 }
739                 return legacySuite;
740             }
741 
742             // no cipher suites in common
743             return null;
744         }

1313                         Authenticator.valueOf(chc.negotiatedProtocol),
1314                         chc.negotiatedProtocol, readKey, readIv,
1315                         chc.sslContext.getSecureRandom());
1316             } catch (GeneralSecurityException gse) {
1317                 // unlikely
1318                 throw chc.conContext.fatal(Alert.HANDSHAKE_FAILURE,
1319                         "Missing cipher algorithm", gse);
1320             }
1321 
1322             if (readCipher == null) {
1323                 throw chc.conContext.fatal(Alert.ILLEGAL_PARAMETER,
1324                     "Illegal cipher suite (" + chc.negotiatedCipherSuite +
1325                     ") and protocol version (" + chc.negotiatedProtocol +
1326                     ")");
1327             }
1328 
1329             chc.baseReadSecret = readSecret;
1330             chc.conContext.inputRecord.changeReadCiphers(readCipher);
1331 
1332             // update the handshake traffic write keys.
1333             SecretKey writeSecret = secretKD.deriveKey(
1334                     "TlsClientHandshakeTrafficSecret", null);
1335             SSLKeyDerivation writeKD =
1336                     kdg.createKeyDerivation(chc, writeSecret);
1337             SecretKey writeKey = writeKD.deriveKey(
1338                     "TlsKey", null);
1339             SecretKey writeIvSecret = writeKD.deriveKey(
1340                     "TlsIv", null);
1341             IvParameterSpec writeIv =
1342                     new IvParameterSpec(writeIvSecret.getEncoded());
1343             SSLWriteCipher writeCipher;
1344             try {
1345                 writeCipher =
1346                     chc.negotiatedCipherSuite.bulkCipher.createWriteCipher(
1347                         Authenticator.valueOf(chc.negotiatedProtocol),
1348                         chc.negotiatedProtocol, writeKey, writeIv,
1349                         chc.sslContext.getSecureRandom());
1350             } catch (GeneralSecurityException gse) {
1351                 // unlikely
1352                 throw chc.conContext.fatal(Alert.HANDSHAKE_FAILURE,
1353                         "Missing cipher algorithm", gse);
1354             }
1355 
1356             if (writeCipher == null) {
1357                 throw chc.conContext.fatal(Alert.ILLEGAL_PARAMETER,
1358                     "Illegal cipher suite (" + chc.negotiatedCipherSuite +
1359                     ") and protocol version (" + chc.negotiatedProtocol +
1360                     ")");
1361             }
1362 
1363             chc.baseWriteSecret = writeSecret;
1364             chc.conContext.outputRecord.changeWriteCiphers(
1365                     writeCipher, (serverHello.sessionId.length() != 0));
1366 
1367             // Should use resumption_master_secret for TLS 1.3.
1368             // chc.handshakeSession.setMasterSecret(masterSecret);
1369 
1370             // Update the context for master key derivation.
1371             chc.handshakeKeyDerivation = secretKD;
1372 
1373             // update the consumers and producers
1374             //
1375             // The server sends a dummy change_cipher_spec record immediately
1376             // after its first handshake message.  This may either be after a
1377             // ServerHello or a HelloRetryRequest.
1378             chc.conContext.consumers.putIfAbsent(
1379                     ContentType.CHANGE_CIPHER_SPEC.id,
1380                     ChangeCipherSpec.t13Consumer);
1381 
1382             chc.handshakeConsumers.put(
1383                     SSLHandshake.ENCRYPTED_EXTENSIONS.id,
1384                     SSLHandshake.ENCRYPTED_EXTENSIONS);
1385 
1386             // Support cert authentication only, when not PSK.
1387             chc.handshakeConsumers.put(
1388                     SSLHandshake.CERTIFICATE_REQUEST.id,
1389                     SSLHandshake.CERTIFICATE_REQUEST);
1390             chc.handshakeConsumers.put(
1391                     SSLHandshake.CERTIFICATE.id,
1392                     SSLHandshake.CERTIFICATE);
1393             chc.handshakeConsumers.put(
1394                     SSLHandshake.CERTIFICATE_VERIFY.id,
1395                     SSLHandshake.CERTIFICATE_VERIFY);
1396 
1397             chc.handshakeConsumers.put(
1398                     SSLHandshake.FINISHED.id,
1399                     SSLHandshake.FINISHED);
1400 
1401             //
1402             // produce
1403             //
1404             // Need no new handshake message producers here.
1405         }

1408     private static final
1409             class T13HelloRetryRequestConsumer implements HandshakeConsumer {
1410         // Prevent instantiation of this class.
1411         private T13HelloRetryRequestConsumer() {
1412             // blank
1413         }
1414 
1415         @Override
1416         public void consume(ConnectionContext context,
1417                 HandshakeMessage message) throws IOException {
1418             // The consuming happens in client side only.
1419             ClientHandshakeContext chc = (ClientHandshakeContext)context;
1420             ServerHelloMessage helloRetryRequest = (ServerHelloMessage)message;
1421             if (helloRetryRequest.serverVersion != ProtocolVersion.TLS12) {
1422                 throw chc.conContext.fatal(Alert.PROTOCOL_VERSION,
1423                     "The HelloRetryRequest.legacy_version is not TLS 1.2");
1424             }
1425 
1426             chc.negotiatedCipherSuite = helloRetryRequest.cipherSuite;
1427 
1428             //
1429             // validate
1430             //
1431 
1432             // Check and launch ClientHello extensions.
1433             SSLExtension[] extTypes = chc.sslConfig.getEnabledExtensions(
1434                     SSLHandshake.HELLO_RETRY_REQUEST);
1435             helloRetryRequest.extensions.consumeOnLoad(chc, extTypes);
1436 
1437             //
1438             // update
1439             //
1440             helloRetryRequest.extensions.consumeOnTrade(chc, extTypes);
1441 
1442             // Change client/server handshake traffic secrets.
1443             // Refresh handshake hash
1444             chc.handshakeHash.finish();     // reset the handshake hash
1445 
1446             // calculate the transcript hash of the 1st ClientHello message
1447             HandshakeOutStream hos = new HandshakeOutStream(null);
1448             try {
1449                 chc.initialClientHelloMsg.write(hos);
1450             } catch (IOException ioe) {
1451                 // unlikely
1452                 throw chc.conContext.fatal(Alert.HANDSHAKE_FAILURE,
1453                     "Failed to construct message hash", ioe);
1454             }
1455             chc.handshakeHash.deliver(hos.toByteArray());
1456             chc.handshakeHash.determine(
1457                     chc.negotiatedProtocol, chc.negotiatedCipherSuite);
1458             byte[] clientHelloHash = chc.handshakeHash.digest();
1459 
1460             // calculate the message_hash
1461             //
1462             // Transcript-Hash(ClientHello1, HelloRetryRequest, ... Mn) =
1463             //   Hash(message_hash ||    /* Handshake type */
1464             //     00 00 Hash.length ||  /* Handshake message length (bytes) */
1465             //     Hash(ClientHello1) || /* Hash of ClientHello1 */
1466             //     HelloRetryRequest || ... || Mn)
1467             int hashLen = chc.negotiatedCipherSuite.hashAlg.hashLength;
1468             byte[] hashedClientHello = new byte[4 + hashLen];
1469             hashedClientHello[0] = SSLHandshake.MESSAGE_HASH.id;
1470             hashedClientHello[1] = (byte)0x00;
1471             hashedClientHello[2] = (byte)0x00;
1472             hashedClientHello[3] = (byte)(hashLen & 0xFF);
1473             System.arraycopy(clientHelloHash, 0,
1474                     hashedClientHello, 4, hashLen);
1475 
1476             chc.handshakeHash.finish();     // reset the handshake hash
1477             chc.handshakeHash.deliver(hashedClientHello);
1478 
1479             int hrrBodyLen = helloRetryRequest.handshakeRecord.remaining();
1480             byte[] hrrMessage = new byte[4 + hrrBodyLen];
1481             hrrMessage[0] = SSLHandshake.HELLO_RETRY_REQUEST.id;
1482             hrrMessage[1] = (byte)((hrrBodyLen >> 16) & 0xFF);
1483             hrrMessage[2] = (byte)((hrrBodyLen >> 8) & 0xFF);
1484             hrrMessage[3] = (byte)(hrrBodyLen & 0xFF);
1485 
1486             ByteBuffer hrrBody = helloRetryRequest.handshakeRecord.duplicate();
1487             hrrBody.get(hrrMessage, 4, hrrBodyLen);
1488 
1489             chc.handshakeHash.receive(hrrMessage);
1490 
1491             // Update the initial ClientHello handshake message.
1492             chc.initialClientHelloMsg.extensions.reproduce(chc,
1493                     new SSLExtension[] {
1494                         SSLExtension.CH_COOKIE,
1495                         SSLExtension.CH_KEY_SHARE,
1496                         SSLExtension.CH_PRE_SHARED_KEY
1497                     });
1498 
1499             //
1500             // produce response handshake message
1501             //
1502             SSLHandshake.CLIENT_HELLO.produce(context, helloRetryRequest);
1503         }

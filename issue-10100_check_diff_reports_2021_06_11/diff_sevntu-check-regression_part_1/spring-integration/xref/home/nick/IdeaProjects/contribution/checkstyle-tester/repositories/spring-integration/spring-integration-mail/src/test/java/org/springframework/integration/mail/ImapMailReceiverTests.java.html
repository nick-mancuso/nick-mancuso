199 	public void testIdleWithServerGuts(ImapMailReceiver receiver, boolean mapped, boolean simple) throws Exception {
200 		receiver.setMaxFetchSize(1);
201 		receiver.setShouldDeleteMessages(false);
202 		receiver.setShouldMarkMessagesAsRead(true);
203 		ThreadPoolTaskScheduler taskScheduler = new ThreadPoolTaskScheduler();
204 		setUpScheduler(receiver, taskScheduler);
205 		receiver.setUserFlag("testSIUserFlag");
206 		receiver.afterPropertiesSet();
207 		ImapIdleChannelAdapter adapter = new ImapIdleChannelAdapter(receiver);
208 		QueueChannel channel = new QueueChannel();
209 		adapter.setOutputChannel(channel);
210 		adapter.setTaskScheduler(taskScheduler);
211 		adapter.setReconnectDelay(1);
212 		adapter.start();
213 		MimeMessage message =
214 				GreenMailUtil.createTextEmail("Foo <foo@bar>", "Bar <bar@baz>", "Test Email", "foo\r\n",
215 						imapIdleServer.getImap().getServerSetup());
216 		message.setRecipients(Message.RecipientType.CC, "a@b, c@d");
217 		message.setRecipients(Message.RecipientType.BCC, "e@f, g@h");
218 		user.deliver(message);
219 		user.deliver(GreenMailUtil.createTextEmail("to", "Bar <bar@baz>", "subject", "body",
220 				imapIdleServer.getImap().getServerSetup()));
221 		if (!mapped) {
222 			@SuppressWarnings("unchecked")
223 			org.springframework.messaging.Message<MimeMessage> received =
224 					(org.springframework.messaging.Message<MimeMessage>) channel.receive(10000);
225 			assertThat(received).isNotNull();
226 			assertThat(received.getPayload().getReceivedDate()).isNotNull();
227 			assertThat(received.getPayload().getLineCount() > -1).isTrue();
228 			if (simple) {
229 				assertThat(received.getPayload().getContent())
230 						.isEqualTo("foo\r\n");
231 			}
232 			else {
233 				assertThat(received.getPayload().getContent())
234 						.isEqualTo("foo");
235 			}
236 		}
237 		else {
238 			org.springframework.messaging.Message<?> received = channel.receive(10000);
239 			assertThat(received).isNotNull();
240 			MessageHeaders headers = received.getHeaders();
241 			assertThat(headers.get(MailHeaders.RAW_HEADERS)).isNotNull();
242 			assertThat(headers.get(MailHeaders.CONTENT_TYPE)).isEqualTo("TEXT/PLAIN; charset=us-ascii");
243 			assertThat(headers.get(MessageHeaders.CONTENT_TYPE)).isEqualTo(MimeTypeUtils.TEXT_PLAIN_VALUE);
244 			assertThat(headers.get(MailHeaders.FROM)).isEqualTo("Bar <bar@baz>");
245 			String[] toHeader = headers.get(MailHeaders.TO, String[].class);
246 			assertThat(toHeader).isNotEmpty();
247 			assertThat(toHeader[0]).isEqualTo("Foo <foo@bar>");
248 			assertThat(Arrays.toString(headers.get(MailHeaders.CC, String[].class))).isEqualTo("[a@b, c@d]");
249 			assertThat(Arrays.toString(headers.get(MailHeaders.BCC, String[].class))).isEqualTo("[e@f, g@h]");
250 			assertThat(headers.get(MailHeaders.SUBJECT)).isEqualTo("Test Email");
251 			if (simple) {
252 				assertThat(received.getPayload()).isEqualTo("foo\r\n");
253 			}
254 			else {
255 				assertThat(received.getPayload()).isEqualTo("foo");
256 			}
257 		}
258 		assertThat(channel.receive(20000)).isNotNull(); // new message after idle
259 		assertThat(channel.receive(100)).isNull(); // no new message after second and third idle
260 
261 		adapter.stop();
262 		taskScheduler.shutdown();
263 		assertThat(imapSearches.stores.get(0)).contains("testSIUserFlag");
264 	}

17  package org.springframework.integration.annotation;
18  
19  import java.lang.annotation.Documented;
20  import java.lang.annotation.ElementType;
21  import java.lang.annotation.Retention;
22  import java.lang.annotation.RetentionPolicy;
23  import java.lang.annotation.Target;
24  
25  import org.springframework.messaging.handler.annotation.ValueConstants;
26  
27  /**
28   * Indicates that a method is capable of transforming a message, message header,
29   * or message payload.
30   *
31   * @author Mark Fisher
32   * @author Gary Russell
33   * @author Artem Bilan
34   */
35  @Target({ ElementType.METHOD, ElementType.ANNOTATION_TYPE })
36  @Retention(RetentionPolicy.RUNTIME)
37  @Documented
38  public @interface Transformer {
39  
40  	/**
41  	 * Specify the channel from which this transformer will consume messages.
42  	 * If the channel does not exist, a {@code DirectChannel} with this name will be
43  	 * registered in the application context.
44  	 * @return The channel name.
45  	 */
46  	String inputChannel() default "";
47  
48  	/**
49  	 * Specify the channel to which this transformer will send the transformed message.
50  	 * @return The channel name.
51  	 */
52  	String outputChannel() default "";
53  
54  	/**
55  	 * Specify a "chain" of {@code Advice} objects that will "wrap" the message handler.
56  	 * Only the handler is advised, not the downstream flow.
57  	 * @return the advice chain.
58  	 */
59  	String[] adviceChain() default { };
60  
61  	/**
62  	 * Specify the maximum amount of time in milliseconds to wait when sending a reply
63  	 * {@link org.springframework.messaging.Message} to the {@code outputChannel}.
64  	 * Defaults to {@code -1} - blocking indefinitely.
65  	 * It is applied only if the output channel has some 'sending' limitations, e.g.
66  	 * {@link org.springframework.integration.channel.QueueChannel} with
67  	 * fixed a 'capacity'. In this case a {@link org.springframework.messaging.MessageDeliveryException} is thrown.
68  	 * The 'sendTimeout' is ignored in case of
69  	 * {@link org.springframework.integration.channel.AbstractSubscribableChannel} implementations.
70  	 * Can be specified as 'property placeholder', e.g. {@code ${spring.integration.sendTimeout}}.
71  	 * @return The timeout for sending results to the reply target (in milliseconds)
72  	 */
73  	String sendTimeout() default "";
74  
75  	/**
76  	 * The {@link org.springframework.context.SmartLifecycle} {@code autoStartup} option.
77  	 * Can be specified as 'property placeholder', e.g. {@code ${foo.autoStartup}}.
78  	 * Defaults to {@code true}.
79  	 * @return the auto startup {@code boolean} flag.
80  	 */
81  	String autoStartup() default "";
82  
83  	/**
84  	 * Specify a {@link org.springframework.context.SmartLifecycle} {@code phase} option.
85  	 * Defaults {@code Integer.MAX_VALUE / 2} for {@link org.springframework.integration.endpoint.PollingConsumer}
86  	 * and {@code Integer.MIN_VALUE} for {@link org.springframework.integration.endpoint.EventDrivenConsumer}.
87  	 * Can be specified as 'property placeholder', e.g. {@code ${foo.phase}}.
88  	 * @return the {@code SmartLifecycle} phase.
89  	 */
90  	String phase() default "";
91  
92  	/**
93  	 * @return the {@link Poller} options for a polled endpoint
94  	 * ({@link org.springframework.integration.scheduling.PollerMetadata}).
95  	 * This attribute is an {@code array} just to allow an empty default (no poller).
96  	 * Mutually exclusive with {@link #reactive()}.
97  	 */
98  	Poller[] poller() default { };
99  
100 	/**
101 	 * @return the {@link Reactive} marker for a consumer endpoint.
102 	 * Mutually exclusive with {@link #poller()}.
103 	 * @since 5.5
104 	 */
105 	Reactive reactive() default @Reactive(ValueConstants.DEFAULT_NONE);
106 
107 }

50  /**
51   * A {@link MessageProducerSupport} for reading messages from a Redis Stream and publishing them into the provided
52   * output channel.
53   * By default this adapter reads message as a standalone client {@code XREAD} (Redis command) but can be switched to a
54   * Consumer Group feature {@code XREADGROUP} by setting {@link #consumerName} field.
55   * By default the Consumer Group name is the id of this bean {@link #getBeanName()}.
56   *
57   * @author Attoumane Ahamadi
58   * @author Artem Bilan
59   * @author Rohan Mukesh
60   *
61   * @since 5.4
62   */
63  public class ReactiveRedisStreamMessageProducer extends MessageProducerSupport {

268 	@Override
269 	protected void onInit() {
270 		super.onInit();
271 		if (this.streamReceiverOptions == null) {
272 			this.streamReceiverOptions = this.streamReceiverOptionsBuilder.build();
273 		}
274 		this.streamReceiver = StreamReceiver.create(this.reactiveConnectionFactory, this.streamReceiverOptions);
275 		if (StringUtils.hasText(this.consumerName) && !StringUtils.hasText(this.consumerGroup)) {
276 			this.consumerGroup = getBeanName();
277 		}
278 		ReactiveRedisTemplate<String, ?> reactiveRedisTemplate =
279 				new ReactiveRedisTemplate<>(this.reactiveConnectionFactory, RedisSerializationContext.string());
280 		this.reactiveStreamOperations = reactiveRedisTemplate.opsForStream();
281 	}
282 
283 	@Override
284 	protected void doStart() {
285 		StreamOffset<String> offset = StreamOffset.create(this.streamKey, this.readOffset);
286 
287 		Flux<? extends Record<String, ?>> events;
288 
289 		if (!StringUtils.hasText(this.consumerName)) {
290 			events = this.streamReceiver.receive(offset);
291 		}
292 		else {
293 			Mono<?> consumerGroupMono = Mono.empty();
294 			if (this.createConsumerGroup) {
295 				consumerGroupMono =
296 						this.reactiveStreamOperations.createGroup(this.streamKey, this.consumerGroup) // NOSONAR
297 								.onErrorReturn(this.consumerGroup);
298 			}
299 
300 			Consumer consumer = Consumer.from(this.consumerGroup, this.consumerName); // NOSONAR
301 
302 			if (offset.getOffset().equals(ReadOffset.latest())) {
303 				// for consumer group offset id should be equal to '>'
304 				offset = StreamOffset.create(this.streamKey, ReadOffset.lastConsumed());
305 			}
306 
307 			events =
308 					this.autoAck
309 							? this.streamReceiver.receiveAutoAck(consumer, offset)
310 							: this.streamReceiver.receive(consumer, offset);
311 
312 			events = consumerGroupMono.thenMany(events);
313 
314 		}
315 
316 		Flux<? extends Message<?>> messageFlux =
317 				events.map((record) -> buildMessageFromRecord(record, this.extractPayload));
318 		subscribeToPublisher(messageFlux);
319 	}
320 
321 	private Message<?> buildMessageFromRecord(Record<String, ?> record, boolean extractPayload) {
322 		AbstractIntegrationMessageBuilder<?> builder =
323 				getMessageBuilderFactory()
324 						.withPayload(extractPayload ? record.getValue() : record)
325 						.setHeader(RedisHeaders.STREAM_KEY, record.getStream())
326 						.setHeader(RedisHeaders.STREAM_MESSAGE_ID, record.getId())
327 						.setHeader(RedisHeaders.CONSUMER_GROUP, this.consumerGroup)
328 						.setHeader(RedisHeaders.CONSUMER, this.consumerName);
329 
330 		if (!this.autoAck && this.consumerGroup != null) {
331 			builder.setHeader(IntegrationMessageHeaderAccessor.ACKNOWLEDGMENT_CALLBACK,
332 					(SimpleAcknowledgment) () ->
333 							this.reactiveStreamOperations
334 									.acknowledge(this.consumerGroup, record)
335 									.subscribe());
336 		}
337 
338 		return builder.build();
339 	}
340 
341 	private <T> Publisher<T> handleReceiverError(Throwable error) {
342 		Message<?> failedMessage = null;
343 		if (error instanceof ConversionFailedException) {
344 			@SuppressWarnings("unchecked")
345 			Record<String, ?> record = (Record<String, ?>) ((ConversionFailedException) error).getValue();
346 			if (record != null) {
347 				failedMessage = buildMessageFromRecord(record, false);
348 			}
349 		}
350 		MessagingException conversionException =
351 				new MessageConversionException(failedMessage, // NOSONAR
352 						"Cannot deserialize Redis Stream Record", error);
353 		if (!sendErrorMessageIfNecessary(null, conversionException)) {
354 			logger.getLog().error(conversionException);
355 		}
356 		return Mono.empty();
357 	}
